


  upstream node_cluster {
    server your_private_ip:3000               weight=2 max_fails=3 fail_timeout=30s;
    server your_private_ip:3001               max_fails=3 fail_timeout=30s;
    server your_private_ip:3002               max_fails=3 fail_timeout=30s;
    keepalive 16;
}

  
  server {    # change you domain_name to yours
    listen 443 ssl http2;
    server_name domain_name;

    ssl_certificate /etc/letsencrypt/live/domain_name/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/domain_name/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Secure Headers
    add_header X-Content-Type-Options nosniff;
    add_header X-Frame-Options DENY;
    add_header X-XSS-Protection "1; mode=block";
    add_header Referrer-Policy no-referrer;
    add_header Permissions-Policy "geolocation=(), microphone=(), camera=()";
    add_header Strict-Transport-Security "max-age=63072000; includeSubDomains; preload" always;

   # This is for the amdin ( admin.html)
  location /admin  {

        allow private_ip;  # Replace with your allowed IP. !!! IMPORTANT !!! The admin panel will be blocked for everyone excepet the ips you allow. YOU HAVE TO USE THE  #PRIVATE IP OF THE SERVER TO HAVE ACCESS TO THE ADMIN FORM THE ALLOWED IPS.ACCESSING THE ADMIN FROM THE DNS YOU GOT,IT WILL BLOCK IT EVEN USING ALLOWED IPS.ONLY #FROM THE SERVER IP WILL WORK.if you want to access it from your domain and not from the private ip and still keep your access rules go and edit the /etc/hosts from the HOST MACHINE AND NOT FROM THE SERVER and add " server_ip_and_not_host_ip  your_domain_name " . Also if your router supports it you can add " server_ip_and_not_host_ip  your_domain_name  " in the dns settings of your router. This way you don't have to make changes to /etc/hosts of each client host .it will take please instantly.
        allow private_ip;
        allow private_ip;
        deny all;


        proxy_pass  http://node_cluster;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;

    }
   # This is for the client ( index.html )
    location / {
        proxy_pass  http://node_cluster;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;

        # Prevent access to hidden files
        location ~ /\.(?!well-known).* {
            deny all;
        }
    }
}

server {
    listen 80;
    server_name domain_name;

    return 301 https://$host$request_uri;
}
